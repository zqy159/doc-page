Axios 拦截器 ​

拦截器封装 ​

避免重复请求: 请求通过拦截器存储 url,响应时删除。再次请求时,判断是否有重复的 URL,如果有终止第一次的请求

js
import axios from "axios";

// 创建Axios实例
const instance = axios.create({
  // baseURL:""
  timeout: 10000, // 超时时间
  headers: {
    "Content-Type": "application/json", // 默认Content-Type
  },
});
const generateReqKey = (config) => {
  const { method, url, params, data } = config;
  return [url, method, JSON.stringify(params), JSON.stringify(data)].join("&");
};
// 存储活动请求的 Map（key: 请求标识, value: 取消函数）
const pendingRequests = new Map();

// 添加请求拦截器
instance.interceptors.request.use(
  (config) => {
    // 1. 添加全局提示（需配合UI库实现）
    // 2. 特定参数控制请求头
    // 3. 统一处理POST参数（如FormData转换）
    // if (
    //   config.method === "post" &&
    //   config.headers["Content-Type"] === "application/json"
    // ) {
    //   // 可选：自动转换JSON数据（根据需求调整）
    // }
    // 4.重复请求处理
    const reqKey = generateReqKey(config);
    if (pendingRequests.has(reqKey)) {
      const cancelToken = pendingRequests.get(reqKey);
      cancelToken("重复请求已取消"); // 取消前一个请求
      pendingRequests.delete(reqKey); // 清理记录
    }
    // 创建新的取消令牌
    config.cancelToken = new axios.CancelToken((cancel) => {
      pendingRequests.set(reqKey, cancel);
    });

    return config;
  },
  (error) => Promise.reject(error)
);

// 添加响应拦截器
instance.interceptors.response.use(
  (response) => {
    // 1. 添加全局提示（需配合UI库实现）
    // 2. 请求成功统一数据处理
    // const res = response.data;
    // if (res.code !== 200) {
    //   // 错误处理（如401跳转登录）
    //   // handleErrorCode(res.code);
    //   return Promise.reject(new Error(res.message || "Error"));
    // }
    // 3.请求成功删除已存储的url
    const reqKey = generateReqKey(response.config);
    pendingRequests.delete(reqKey); // 响应成功后清理
    // return response;
    return res.data; // 返回核心数据
  },
  (error) => {
    // 1.统一错误处理
    if (axios.isCancel(error)) {
      console.log("请求已取消:", error.message);
    } else {
      const reqKey = generateReqKey(error.config || {});
      pendingRequests.delete(reqKey); // 错误时也需清理
    }
    return Promise.reject(error);
  }
);

// 导出实例
export default instance;
