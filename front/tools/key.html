生成密钥

js
const crypto = require('crypto');
const fs = require('fs');
const path = require('path');
 
// 生成2048位RSA密钥对
const { publicKey, privateKey } = crypto.generateKeyPairSync('rsa', {
  modulusLength: 2048,
  publicKeyEncoding: {
    type: 'spki',
    format: 'pem'
  },
  privateKeyEncoding: {
    type: 'pkcs8',
    format: 'pem',
    cipher: 'aes-256-cbc',
    passphrase: 'StrongPassphrase123!' // 生产环境建议使用环境变量
  }
});
 
// 保存密钥到文件（实际应存入密钥管理系统）
fs.writeFileSync(path.join(__dirname, 'rsa_public.pem'), publicKey);
fs.writeFileSync(path.join(__dirname, 'rsa_private.pem'), privateKey);
 
  
console.log('Public Key:\n', publicKey);  
console.log('Private Key:\n', privateKey);

前端加密

https://cdn.jsdelivr.net/npm/jsencrypt/bin/jsencrypt.min.js

js
<script>
// 2. 从后端获取的公钥（实际应通过安全通道获取）

const PUBLIC_KEY = `-----BEGIN PUBLIC KEY-----
YOUR_PUBLIC_KEY_HERE
-----END PUBLIC KEY-----`;

// 3. RSA加密函数
function encryptData(data) {
  const encryptor = new JSEncrypt();
  encryptor.setPublicKey(PUBLIC_KEY);
  
  // 处理长文本分段加密（RSA-OAEP限制）
  const MAX_ENCRYPT_BLOCK = 190; // 2048位密钥最大加密块
  let encrypted = '';
  
  for (let i = 0; i < data.length; i += MAX_ENCRYPT_BLOCK) {
    const chunk = data.substr(i, MAX_ENCRYPT_BLOCK);
    encrypted += encryptor.encrypt(chunk, false, 'base64') + '.';
  }
  
  return encrypted.slice(0, -1); // 移除最后一个分隔符
}

// 使用示例
const sensitiveData = '用户敏感信息123!@#';
const encrypted = encryptData(sensitiveData);
console.log('加密结果:', encrypted);
</script>

后端解密

js
const crypto = require('crypto');
const fs = require('fs');

// 读取私钥（实际应从安全存储获取）
const PRIVATE_KEY = fs.readFileSync('./rsa_private.pem', 'utf8');

function decryptData(encryptedData) {
  const chunks = encryptedData.split('.');
  let decrypted = '';
  
  for (const chunk of chunks) {
    const buffer = Buffer.from(chunk, 'base64');
    const decryptedChunk = crypto.privateDecrypt({
      key: PRIVATE_KEY,
      passphrase: 'StrongPassphrase123!', // 与生成密钥时一致
      padding: crypto.constants.RSA_PKCS1_OAEP_PADDING,
      oaepHash: 'sha256'
    }, buffer);
    
    decrypted += decryptedChunk.toString('utf8');
  }
  
  return decrypted;
}

// 使用示例（需替换为实际接收的加密数据）
const encryptedPayload = '前端加密后的数据';
const originalData = decryptData(encryptedPayload);
console.log('解密结果:', originalData);
